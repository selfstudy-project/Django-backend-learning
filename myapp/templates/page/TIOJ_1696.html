const int maxn = 1e5 + 5 ;
const int mod = 1e9 + 7 ;
const int INF = 1e9 + 9 ;
vector<int>g[maxn];
int k[maxn],sz[maxn],cnt[maxn+5],ans[maxn],n;
bool del[maxn];
void add(int x,int v) {
	x++;
	while(x<=n+1) {
		cnt[x]+=v;
		x+=lowbit(x);
	}
}
int cal(int x,int n) {
	if(x>n) x=n;
	int ret=0;
	x++;
	while(x>0) {
		ret+=cnt[x];
		x-=lowbit(x);
	}
	return ret;
}
void get_sz(int x,int p) {
	sz[x]=1;
	for(auto i:g[x]) {
		if(i==p or del[i]) continue;
		get_sz(i,x);
		sz[x]+=sz[i];
	}
}
int get_cen(int x,int n,int p) {
	for(auto i:g[x]) {
		if(i==p or del[i]) continue;
		if(sz[i]>n/2) return get_cen(i,n,x);
	}
	return x;
}
vector<pii>v;
void dfs(int x,int p,int level) {
	add(level,1);
	v.pb({x,level});
	for(auto i:g[x]) {
		if(i==p or del[i]) continue;
		dfs(i,x,level+1);
	}
}
void build(int x) {
	get_sz(x,-1);
	int cen=get_cen(x,sz[x],-1);
	del[cen]=true;
	fill(cnt,cnt+sz[x]+5,0);
	vector<tup>vet;
	for(auto i:g[cen]) {
		if(del[i]) continue;
		dfs(i,cen,1);
		for(auto i:v) vet.pb({i.f,cal(k[i.f]-i.s,sz[x]),i.s});
		while(!v.empty()) {
			auto i=v.back();
			add(i.s,-1);
			v.pop_back();
		}
	}
	add(0,1);
	for(auto [a,b,c]:vet) add(c,1); 
	for(auto [u,val,level]:vet) {
		ans[u]+=cal(k[u]-level,sz[x])-val;
	}
	ans[cen]+=cal(k[cen],sz[x]);
	for(auto i:g[cen]) {
		if(del[i]) continue;
		build(i);
	}
}
